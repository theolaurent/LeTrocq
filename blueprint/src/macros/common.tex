% Macros and settings shared by BOTH the pdf and web builds.

% --- theorem-like environments (these are the nodes of the dependency graph) ---
% All share the `theorem` counter (the [theorem] argument) and never reset it.
\newtheorem{theorem}{Theorem}
\newtheorem{proposition}[theorem]{Proposition}
\newtheorem{lemma}[theorem]{Lemma}
\newtheorem{corollary}[theorem]{Corollary}

\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition}

\theoremstyle{remark}
\newtheorem{remark}[theorem]{Remark}

% --- notation. Kept MathJax-safe (no stmaryrd / mathpartir) so the web renderer copes.
%     Every macro is wrapped in \ensuremath so it works in text mode too (these names
%     appear both inline in prose and inside math). ---

% the two objects of the hierarchy
\newcommand{\Param}{\ensuremath{\mathsf{Param}}}
\newcommand{\MapHas}{\ensuremath{\mathsf{MapHas}}}
\newcommand{\MapClass}{\ensuremath{\mathsf{MapClass}}}

% sorts
\newcommand{\Type}{\ensuremath{\mathsf{Type}}}
\newcommand{\PropSort}{\ensuremath{\mathsf{Prop}}}
\newcommand{\SortU}{\ensuremath{\mathsf{Sort}}}

% the twofold translation
\newcommand{\Cpt}[1]{\ensuremath{\langle #1 \rangle}}      % counterpart  ⟨t⟩
\newcommand{\Rel}[1]{\ensuremath{[ #1 ]}}                   % relatedness  [t]
\newcommand{\Sem}[1]{\ensuremath{[\![ #1 ]\!]}}           % relation     〚A〛

% classes and judgments
\newcommand{\cls}[2]{\ensuremath{(#1,#2)}}                 % a ParamClass pair
\newcommand{\rw}{\ensuremath{\rightsquigarrow}}            % "produces"   ↝
\newcommand{\wk}{\ensuremath{\Downarrow}}                   % weakening    ⇓
\newcommand{\chk}{\ensuremath{\Leftarrow}}                  % check mode
\newcommand{\syn}{\ensuremath{\Rightarrow}}                 % infer/leaf mode

% combinators (names as they read in the prose)
\newcommand{\RArrow}{\ensuremath{\mathsf{RArrow}}}
\newcommand{\RForall}{\ensuremath{\mathsf{RForall}}}

% a two-premise inference rule that renders under MathJax and xelatex alike
\newcommand{\irule}[2]{\dfrac{\begin{array}{c}#1\end{array}}{#2}}
